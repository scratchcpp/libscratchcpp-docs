38inline constexpr Rgb rgb(int r, int g, int b)
39{
40 return (0xffu << 24) | ((r & 0xffu) << 16) | ((g & 0xffu) << 8) | (b & 0xffu);
41}

44inline constexpr Rgb rgba(int r, int g, int b, int a)
45{
46 return ((a & 0xffu) << 24) | ((r & 0xffu) << 16) | ((g & 0xffu) << 8) | (b & 0xffu);
47}
