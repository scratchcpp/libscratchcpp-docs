147 inline void toString(std::string &dst) const
148 {
149 dst.clear();
150 veque::veque<std::string> strings;
151 strings.reserve(m_size);
152 bool digits = true;
153 size_t i;
154
155 for (i = 0; i < m_size; i++) {
156 const ValueData *item = &m_dataPtr->operator[](i);
157 strings.push_back(std::string());
158 value_toString(item, &strings.back());
159
160 if (value_isValidNumber(item) && !value_isBool(item) && !strings.back().empty()) {
161 double doubleNum = value_toDouble(item);
162 long num = value_toLong(item);
163
164 if (doubleNum != num) {
165 digits = false;
166 break;
167 }
168
169 if (num < 0 || num >= 10) {
170 digits = false;
171 break;
172 }
173 } else {
174 digits = false;
175 break;
176 }
177 }
178
179 std::string s;
180
181 if (digits) {
182 for (i = 0; i < strings.size(); i++)
183 dst.append(strings[i]);
184
185 for (; i < m_size; i++) {
186 value_toString(&m_dataPtr->operator[](i), &s);
187 dst.append(s);
188 }
189 } else {
190 for (i = 0; i < strings.size(); i++) {
191 dst.append(strings[i]);
192
193 if (i + 1 < m_size)
194 dst.push_back(' ');
195 }
196
197 for (; i < m_size; i++) {
198 value_toString(&m_dataPtr->operator[](i), &s);
199 dst.append(s);
200
201 if (i + 1 < m_size)
202 dst.push_back(' ');
203 }
204 }
205 }
